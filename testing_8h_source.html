21 double computeObjective(const Eigen::MatrixXd &H, const Eigen::VectorXd &h, const Eigen::VectorXd &primal)
22 {
23 Eigen::MatrixXd L = H.triangularView<Eigen::Lower>();
24
25 double result = 0.5 * primal.transpose() * L * L.transpose() * primal;
26
27 if (h.rows() > 0)
28 {
29 result += h.transpose() * primal;
30 }
31
32 std::cout << "||| Objective = " << result << std::endl;
33
34 return (result);
35 }

53
54 if (h.rows() > 0)

56 v += h;
57 }

64

67 M.col(i).setZero();
68 switch (constraints_status[ctr_index])

70 case ConstraintStatus::ACTIVE_LOWER_BOUND:

73 case ConstraintStatus::ACTIVE_UPPER_BOUND:
74 case ConstraintStatus::EQUALITY:
75 M(ctr_index, i) = 1.0;
76 break;
77 default:

79 }
80 }
81 else

83 switch (constraints_status[ctr_index])
84 {
85 case ConstraintStatus::ACTIVE_LOWER_BOUND:

87 break;
88 case ConstraintStatus::ACTIVE_UPPER_BOUND:
89 case ConstraintStatus::EQUALITY:
90 M.col(i) = A.row(ctr_index - num_simple_bounds).transpose();
91 break;
92 default:
93 break;
94 }
95 }
96 }

129 {
130 case ConstraintStatus::ACTIVE_LOWER_BOUND:
131 case ConstraintStatus::ACTIVE_UPPER_BOUND:
